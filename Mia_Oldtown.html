Historical Properties Analysis

We define historical properties as those built before 1930. Let’s compare historical and non-historical properties.

oldtown_data <- oldtown_data %>% mutate(Historical = if_else(YearBuilt < 1930, "Historical", "Non-Historical"))
summary_stats <- oldtown_data %>%
  group_by(Historical) %>%
  summarise(
    Avg_SalePrice = mean(SalePrice),
    Median_SalePrice = median(SalePrice),
    Avg_OverallQual = mean(as.numeric(OverallQual)),
    Median_OverallQual = median(as.numeric(OverallQual)),
    Avg_GrLivArea = mean(GrLivArea),
    Median_GrLivArea = median(GrLivArea)
  )

kable(summary_stats, caption = "Summary Statistics for Historical vs Non-Historical Properties")

# Preparing the data for the pie chart
garage_data <- oldtown_data %>%
  group_by(GarageType) %>%
  summarise(Count = n()) %>%
  mutate(Percentage = Count / sum(Count) * 100)

# Creating the pie chart
ggplot(garage_data, aes(x = "", y = Percentage, fill = GarageType)) +
  geom_bar(stat = "identity", width = 1) +
  coord_polar("y") +
  labs(title = "Distribution of Garage Types in Oldtown", x = NULL, y = NULL, fill = "Garage Type") +
  theme_minimal() +
  theme(axis.text = element_blank(), axis.title = element_blank(), panel.grid = element_blank(),
